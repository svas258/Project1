pipeline{
   agent any
   tools {
     maven 'maven'
   }
   stages {
    stage ('compile') {
        steps{
            sh 'mvn compile'
        }
    }
    stage ('Test'){
        steps{
            sh 'mvn test'
        }
    }
    stage ('Build'){
        steps{
            sh 'mvn clean build'
        }
    }
   }
}
